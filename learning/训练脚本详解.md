```python
# data/dataset.py:287+
def create_dataloaders(...):
    # 1. 读取文本文件
    text = open(data_path).read()
    
    # 2. 编码成 token ids
    token_ids = tokenizer.encode(text)   # [45, 23, 67, ...]
    
    # 3. 切分训练/验证集
    split_idx = int(len(token_ids) * (1 - val_split))
    train_ids = token_ids[:split_idx]
    val_ids = token_ids[split_idx:]
    
    # 4. 创建 Dataset
    train_dataset = CausalLMDataset(train_ids, seq_len=256)
    val_dataset = CausalLMDataset(val_ids, seq_len=256)
    
    # 5. 创建 DataLoader
    train_loader = DataLoader(train_dataset, batch_size=4, shuffle=True)
    val_loader = DataLoader(val_dataset, batch_size=4, shuffle=False)
```

**CausalLMDataset 内部**:

```python
# data/dataset.py:219-225
def __init__(self, token_ids, seq_len):
    block_len = seq_len + 1   # 257
    n_blocks = len(token_ids) // block_len
    self.blocks = [
        token_ids[i:i+block_len]
        for i in range(0, len(token_ids), block_len)
    ]

def __getitem__(self, idx):
    block = self.blocks[idx]   # 257个token
    return (
        torch.tensor(block[:-1]),  # input_ids: 前256个
        torch.tensor(block[1:]),   # labels: 后256个
    )
```

### 5.2 Batch 循环

```python

```

**一个 batch 的完整流程（以 batch_idx=0 为例）:**

#### Step 1: 数据搬运

```python
# 行 342-343
input_ids = input_ids.to(device)   # (4, 256) → GPU
labels = labels.to(device)          # (4, 256) → GPU
```

#### Step 2: 更新学习率

```python
# 行 352-355
global_step += 1
lr = compute_lr(global_step, train_cfg, model_cfg)
# compute_lr 内部:
#   step < 800: lr = 3e-4 * (step / 800)   # 线性预热
#   step >= 800: lr = 3e-4                  # 恒定
for pg in optimizer.param_groups:
    pg["lr"] = lr
```

#### Step 3: Forward

```python
# 行 370-375
with torch.autocast(device_type="cuda", dtype=amp_dtype):
    logits = model(input_ids)
# logits: (4, 256, 16000)
```

**model.forward 内部调用链**:

```python
# model/transformer.py:167
def forward(self, input_ids):
    tgt_mask = self.make_causal_mask(input_ids)   # (4, 1, 256, 256)
    hidden = self.decoder(input_ids, tgt_mask)    # (4, 256, 768)
    logits = self.output_proj(hidden)             # (4, 256, 16000)
    return logits
```

#### Step 4: 计算 Loss

```python
# 行 394-401
loss = criterion(
    logits.reshape(-1, logits.size(-1)),   # (1024, 16000)
    labels.reshape(-1)                      # (1024,)
)
# 如果启用 MoE:
if model_cfg.use_moe:
    loss = loss + 0.01 * collect_moe_load_balance_loss(model)
```

**CrossEntropyLoss 内部计算过程：**

```python
# reshape 后:
logits_flat: (1024, 16000)   # 1024个位置，每个位置16000个分数
labels_flat: (1024,)          # 1024个正确答案的token id

# CrossEntropyLoss 内部等价于:
losses = []
for i in range(1024):
    pos_logits = logits_flat[i]     # (16000,) 这个位置的16000个分数
    pos_label = labels_flat[i]      # 正确答案的id
    
    # 1. softmax: 分数变概率
    probs = F.softmax(pos_logits, dim=-1)   # (16000,)
    
    # 2. 取出正确标签对应的概率
    correct_prob = probs[pos_label]
    
    # 3. 负对数
    loss_i = -torch.log(correct_prob)
    losses.append(loss_i)

loss = torch.stack(losses).mean()   # 1024个位置取平均
```

**实际 PyTorch 不用循环，一次矩阵运算完成：**

```python
# F.cross_entropy(logits_flat, labels_flat) 底层:
# 1. 对每行做 softmax
# 2. 取每行正确列的概率
# 3. 取负对数
# 4. 平均
# 全部并行，没有显式循环
```

#### Step 5: 梯度累积归一化

```python
# 行 420
scaled_loss = loss / train_cfg.grad_accum_steps   # loss / 8
```

#### Step 6: Backward

```python
# 行 429-432
if scaler is not None:
    scaler.scale(scaled_loss).backward()
else:
    scaled_loss.backward()
```

**backward 后发生了什么?**

```python
# 每个参数 param 的 grad 被填充
for param in model.parameters():
    # param.grad += d(loss)/d(param)
    pass
```

#### Step 7: 参数更新（每 grad_accum_steps 步触发）

```python
# 行 450-460
should_step = (batch_idx + 1) % train_cfg.grad_accum_steps == 0
# 例如: batch_idx=7, grad_accum_steps=8 → should_step=False
#       batch_idx=7, batch_idx+1=8, 8%8=0 → should_step=True

if should_step:
    # 行 454
    nn.utils.clip_grad_norm_(model.parameters(), train_cfg.max_grad_norm)
    # max_grad_norm=1.0: 如果梯度范数>1.0, 整体缩放到1.0
    
    # 行 456-459
    if scaler is not None:
        scaler.step(optimizer)    # 调用 optimizer.step()
        scaler.update()            # 更新 fp16 缩放因子
    else:
        optimizer.step()           # w = w - lr * grad
    
    # 行 460
    optimizer.zero_grad(set_to_none=True)   # 清空梯度
```

#### Step 8-9: 记录和打印

```python
# 行 465-466
total_loss += loss.item()
n_batches += 1

# 行 477-486
if global_step % train_cfg.log_interval == 0:
    avg = total_loss / n_batches
    log.info(f"第{epoch}轮 | step={global_step} | loss={loss.item():.4f} | 平均={avg:.4f} | lr={lr:.2e}")
```

### 5.3 残余梯度处理

```python
# 行 494-503
if n_batches > 0 and n_batches % train_cfg.grad_accum_steps != 0:
    # 最后一个 epoch 的 batch 数不是 8 的整数倍
    # 例如: 2500 batches, 2500 % 8 = 4, 剩下4个batch的梯度没更新
    nn.utils.clip_grad_norm_(model.parameters(), train_cfg.max_grad_norm)
    optimizer.step()
    optimizer.zero_grad(set_to_none=True)
```

## Phase 6: 验证（evaluate）

**代码位置**: `cli/train.py:512-551`

```python
@torch.no_grad()   # 不计算梯度
def evaluate(model, loader, criterion, device, amp_dtype):
    model.eval()    # 关闭 dropout
    total_loss = 0.0
    n_batches = 0
    
    for input_ids, labels in loader:
        input_ids = input_ids.to(device)
        labels = labels.to(device)
        
        with torch.autocast(device_type="cuda", dtype=amp_dtype):
            logits = model(input_ids)
            loss = criterion(logits.reshape(-1, vocab), labels.reshape(-1))
        
        total_loss += loss.item()
        n_batches += 1
    
    return total_loss / max(n_batches, 1)
```

**和训练的区别**:
- 有 `@torch.no_grad()`: 不创建计算图，省显存
- 有 `model.eval()`: 关闭 dropout
- 没有 `backward()` 和 `optimizer.step()`: 不更新参数
